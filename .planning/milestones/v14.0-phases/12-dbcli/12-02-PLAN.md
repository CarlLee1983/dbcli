<!-- MessageLoader API from Plan 01 -->

**Import and Usage:**
```typescript
import { t, t_vars } from '@/i18n/message-loader'

// Simple message
const welcome = t('init.welcome')

// With interpolation
const success = t_vars('success.inserted', { count: 5 })
// Output: "Successfully inserted 5 row(s)"
```

**Available Message Keys (from resources/lang/en/messages.json):**
- init: welcome, description, select_system, prompt_host, prompt_port, prompt_user, prompt_password, prompt_name, prompt_permission, env_parse_note, connection_testing, connection_success, config_saved, config_exists_overwrite, cancelled
- schema: description, fetching, success, not_found
- list: description, no_tables
- query: description, executing, no_results, result_count
- errors: message, invalid_config, connection_failed, permission_denied, table_not_found, unsupported_lang, invalid_json
- success: inserted, updated, deleted, no_changes
- insert: description, prompt_data, confirm
- update: description, prompt_where, prompt_set, confirm
- delete: description, prompt_where, confirm, admin_only
- export: description, formats, exported
- skill: description, installed

**Command Refactoring Pattern:**
```typescript
import { Command } from 'commander'
import { t, t_vars } from '@/i18n/message-loader'

export const commandName = new Command('name')
  .description(t('namespace.description'))
  .action(async (arg, options) => {
    try {
      console.log(t('namespace.message'))
      // error handling with t_vars()
    } catch (error) {
      console.error(t_vars('errors.message', { message: error.message }))
      process.exit(1)
    }
  })
```

**Test Pattern (for integration tests):**
```typescript
import { test, expect, beforeEach, vi } from 'vitest'
import { t } from '@/i18n/message-loader'

beforeEach(() => { vi.resetModules() })

test('command outputs English by default', async () => {
  vi.stubEnv('DBCLI_LANG', 'en')
  // Capture command output and assert English
})

test('command outputs Chinese when DBCLI_LANG=zh-TW', async () => {
  vi.stubEnv('DBCLI_LANG', 'zh-TW')
  // Capture command output and assert Chinese
})
```
