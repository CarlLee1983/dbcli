```typescript
export type StatementType = 'SELECT' | 'INSERT' | 'UPDATE' | 'DELETE' | 'ALTER' | 'DROP' | 'CREATE' | 'TRUNCATE' | 'SHOW' | 'DESCRIBE' | 'EXPLAIN' | 'UNKNOWN'

  isDangerous: boolean
  keywords: string[]
  isComposite: boolean

  requiredPermission: Permission
}

export function checkPermission(sql: string, permission: Permission): PermissionCheckResult

```

Create src/core/permission-guard.ts with the following functions:

1. **normalizeSQL(sql: string)**: Remove line comments (--), block comments (/* */), compress whitespace, trim
   - Use regex: /--[^\n]*\n/g for line comments
   - Use regex: /\/\*[\s\S]*?\*\//g for block comments
   - Compress spaces with /\s+/g → ' '

2. **stripCommentsAndStrings(sql: string)**: Character-by-character state machine to remove comments AND string literals
   - Handle line comments: -- until \n
   - Handle block comments: /* */ nested handling
   - Handle string literals: 'string' and "string" with escape handling (backslash escapes)
   - Return cleaned SQL (strings replaced with spaces to preserve structure)

3. **detectCompositePatterns(sql: string)**: Return object with hasWithClause, hasSubquery, hasUnion flags
   - WITH clause: /\bWITH\b/ regex on uppercase
   - Subquery: /\(\s*SELECT\b/ regex
   - UNION: /\bUNION\b/ regex
   - Return { hasWithClause: boolean, hasSubquery: boolean, hasUnion: boolean }

4. **extractFirstKeyword(sql: string)**: Find the first SQL keyword (SELECT, INSERT, UPDATE, DELETE, WITH, etc.)
   - Split on whitespace, filter non-empty
   - Skip empty tokens
   - Return uppercase first keyword
   - Handle parameterized queries: strip ? and $N before processing

5. **mapKeywordToType(keyword: string)**: Map SQL keyword to StatementType
   - SELECT, SHOW, DESCRIBE, EXPLAIN → read-only
   - INSERT, UPDATE → write operations
   - DELETE, DROP, ALTER, TRUNCATE, CREATE, GRANT → destructive
   - WITH → recurse (WITH is not a statement type; caller must check outer keyword)
   - Default → UNKNOWN

6. **isDestructiveOperation(type: StatementType)**: Return true if DELETE, DROP, ALTER, TRUNCATE, CREATE, GRANT

7. **extractAllKeywords(sql: string)**: Return array of all SQL keywords (uppercase, deduplicated)

8. **determineConfidence(type: StatementType, keyword: string, sql: string)**: Return HIGH, MEDIUM, or LOW
   - HIGH: Standard SQL operations (SELECT, INSERT, UPDATE, DELETE, WITH, UNION)
   - MEDIUM: Schema operations (CREATE, ALTER, DROP, TRUNCATE)
   - LOW: Vendor-specific or uncommon (VACUUM, OPTIMIZE, CALL, etc.)

9. **removeParameterMarkers(sql: string)**: Strip $1, $2, ? placeholders before analysis
   - Replace /\$\d+/g with ''
   - Replace /\?/g with ''

- class PermissionError extends Error (see task 2)
- function classifyStatement(sql: string): StatementClassification
- Helper functions for testing: normalizeSQL, stripCommentsAndStrings, detectCompositePatterns, extractFirstKeyword, mapKeywordToType

2. **classifyStatement(sql: string)**: Main classification function
   - Call normalizeSQL(sql)
   - Call stripCommentsAndStrings(normalized)
   - Call removeParameterMarkers(stripped)
   - Call detectCompositePatterns(upper)
   - Call extractFirstKeyword(stripped)
   - Call mapKeywordToType(firstKeyword) → type
   - Call isDestructiveOperation(type) → isDangerous
   - Call extractAllKeywords(stripped) → keywords
   - Call determineConfidence(type, firstKeyword, upper) → confidence
   - Return StatementClassification object with all fields

3. **checkPermission(sql: string, permission: Permission)**: Check if operation is allowed
   - Call classifyStatement(sql) → classification
   - If permission === 'admin': return { allowed: true, reason: 'Admin permission: all operations allowed', classification }
   - If permission === 'read-write':
     - allowedTypes = ['SELECT', 'INSERT', 'UPDATE', 'SHOW', 'DESCRIBE', 'EXPLAIN']
     - If classification.type in allowedTypes: return { allowed: true, reason: `{type} operation allowed in read-write mode`, classification }
     - Else: return { allowed: false, reason: `{type} operation requires admin permission`, classification }
   - If permission === 'query-only':
     - allowedTypes = ['SELECT', 'SHOW', 'DESCRIBE', 'EXPLAIN']
     - If classification.type in allowedTypes: return { allowed: true, reason: `{type} operation allowed in query-only mode`, classification }
     - Else: return { allowed: false, reason: `{type} operation requires read-write or admin permission`, classification }
   - Return { allowed: false, reason: `Unknown permission level: {permission}`, classification }

4. **enforcePermission(sql: string, permission: Permission)**: Throws on denial
   - Call checkPermission(sql, permission) → result
   - If !result.allowed: throw new PermissionError(result.reason, result.classification, permission)
   - Else: return result.classification
   - Use this in Phase 6 command handlers before execution
